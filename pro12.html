  • Program to Transpose matrix

    • Input:

    import java.util.Scanner; class transposematrix { public static void main(String[] a) { Scanner s = new Scanner(System.in); System.out.print("Enter number of rows: "); int rows = s.nextInt(); System.out.print("Enter number of columns: "); int columns = s.nextInt(); int[][] matrix = new int[rows][columns]; System.out.println("Enter elements of the matrix:"); for (int i = 0; i < rows; i++) { for (int j = 0; j < columns; j++) { matrix[i][j] = s.nextInt(); } } int[][] transpose = new int[columns][rows]; for (int i = 0; i < rows; i++) { for (int j = 0; j < columns; j++) { transpose[j][i] = matrix[i][j]; } } System.out.println("Original Matrix:"); for (int i = 0; i < rows; i++) { for (int j = 0; j < columns; j++) { System.out.print(matrix[i][j] + " "); } System.out.println(); } System.out.println("Transposed Matrix:"); for (int i = 0; i < columns; i++) { for (int j = 0; j < rows; j++) { System.out.print(transpose[i][j] + " "); } System.out.println(); } } }
